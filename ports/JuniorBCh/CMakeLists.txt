cmake_minimum_required(VERSION 3.12)
project(JuniorBChSC LANGUAGES CXX)
option(SCSYNTH "Build the scsynth plugin target" ON)
option(SUPERNOVA "Build the supernova plugin target" OFF)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
endif()
if(NOT SC_PATH)
    set(SC_PATH "../../supercollider")
endif()
get_filename_component(SC_PATH_FOR_MODULES "${SC_PATH}" ABSOLUTE BASE_DIR "${CMAKE_CURRENT_SOURCE_DIR}")
list(APPEND CMAKE_MODULE_PATH "${SC_PATH_FOR_MODULES}/tools/cmake_gen")
include(SuperColliderServerPlugin)
sc_do_initial_compiler_config()
sc_check_sc_path("${SC_PATH}")
set(juniorbch_cpp_files src/JuniorBCh.cpp)
set(juniorbch_sc_files Classes/JuniorBCh.sc)
set(juniorbch_schelp_files HelpSource/Classes/JuniorBCh.schelp)
sc_add_server_plugin("JuniorBCh" "JuniorBCh" "${juniorbch_cpp_files}" "${juniorbch_sc_files}" "${juniorbch_schelp_files}")
if(TARGET JuniorBCh_scsynth)
    set_target_properties(JuniorBCh_scsynth PROPERTIES OUTPUT_NAME "JuniorBCh")
endif()
install(FILES models/junior_1_stage.json DESTINATION JuniorBCh/models)
