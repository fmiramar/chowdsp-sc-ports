cmake_minimum_required(VERSION 3.12)
project(MetalFaceChSC LANGUAGES CXX)
option(SCSYNTH "Build the scsynth plugin target" ON)
option(SUPERNOVA "Build the supernova plugin target" OFF)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
endif()
if(NOT SC_PATH)
    set(SC_PATH "../../supercollider")
endif()
get_filename_component(SC_PATH_FOR_MODULES "${SC_PATH}" ABSOLUTE BASE_DIR "${CMAKE_CURRENT_SOURCE_DIR}")
list(APPEND CMAKE_MODULE_PATH "${SC_PATH_FOR_MODULES}/tools/cmake_gen")
include(SuperColliderServerPlugin)
sc_do_initial_compiler_config()
sc_check_sc_path("${SC_PATH}")
set(metalfacech_cpp_files src/MetalFaceCh.cpp)
set(metalfacech_sc_files Classes/MetalFaceCh.sc)
set(metalfacech_schelp_files HelpSource/Classes/MetalFaceCh.schelp)
sc_add_server_plugin("MetalFaceCh" "MetalFaceCh" "${metalfacech_cpp_files}" "${metalfacech_sc_files}" "${metalfacech_schelp_files}")
if(TARGET MetalFaceCh_scsynth)
    set_target_properties(MetalFaceCh_scsynth PROPERTIES OUTPUT_NAME "MetalFaceCh")
endif()
install(FILES ../../BYOD/res/guitar_ml_models/metal_face_model.json DESTINATION MetalFaceCh/models)
